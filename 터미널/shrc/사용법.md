```bash

####	공통 사항
#
#
#	코멘트
_ 이 폴더의 [.shrc]는 전부 원본 파일의 심볼릭 링크이므로 수정시 주의
_ 터미널 닫고 재실행하면 [.shrc]가 실행되어 적용됨.
_ 
#
#
#	bashrc 수정
vim ~/.bashrc


####	alias
#
#
#	zz	--->	마지막으로 작업한 폴더로 이동
export PROMPT_COMMAND="export PROMPT_COMMAND='pwd > ~/.last_dir'"
alias zz='cd "$(<~/.last_dir)"'
alias ㅋㅋ='cd "$(<~/.last_dir)"'
#
#
#						nix 통제 없이, 각 폴더의 가상환경(venv)으로 실행.
#	pythonv	--->		터미널에서 python 대신에 사용. pythonv example.py
alias pythonv='./.venv/bin/python'
#
#
#						nix 통제 없이, 파이썬 패키지 관리자 pip 실행.
#	pipv	--->		터미널에서 pip 대신에 사용. pipv install flask
alias pipv="./.venv/bin/pip"
